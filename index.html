    #Write subtracted image to new folder
    cv2.imwrite(dest_path + '/' + file,sub_img)
    min_mse_ball = 10000
    #connected components block / declare constants for connected components
    blur_radius = 5
    threshold = 10
    sqd_img = np.hstack((sub_img,np.zeros((1920,840))))
    img = sqd_img # gray-scale image
    # smooth the image (to remove small objects)
    imgf = ndimage.gaussian_filter(img, blur_radius)
    # find connected components
    labeled, nr_objects = ndimage.label(imgf > threshold)
    #itterate through connected components (cc)
    for i in range(1,nr_objects+1):
      mse_ball = 10000 
      indices = np.indices(img.shape).T[:,:,[1, 0]]
      sub_img = indices[labeled == i]
      x_axis = sub_img[:,1]
      y_axis = sub_img[:,0]
      #verify if cc meets area constraint
      if (len(sub_img[:,1])<(350*350)) and (len(sub_img[:,1])>(40*40)):
          max_heights = []
          for j in range(min(x_axis),max(x_axis)):
              max_heights.append(min(sub_img[sub_img[:,1]==j][:,0]))
          x_fit = [n for n in range(0,len(max_heights))]
          x_fit_arr = np.array(x_fit)
          heights = max_heights-min(max_heights)
          a = np.polyfit(x_fit,-heights, 2)
          ref_fit = ref_coe[0]*(x_fit_arr**2)+ref_coe[1]*x_fit_arr+ref_coe[2]
          if (a[0]!=0):
              #CC APEX
              x_apex = -a[1]/(2*a[0])
              y_apex = a[0]*(x_apex**2)+a[1]*x_apex+a[2]
              #REF APEX
              x_r_apex = -ref_coe[1]/(2*ref_coe[0])
              y_r_apex = ref_coe[0]*(x_r_apex**2)+ref_coe[1]*x_r_apex+ref_coe[2]
              #Shifts
              x_shift = x_r_apex-x_apex
              y_shift = y_r_apex-y_apex
              x_len = max(x_axis)-min(x_axis)
              #create polynomial fit for ball and check against reference ball
              y_fit = a[0]*((x_fit_arr-x_shift)**2)+a[1]*(x_fit_arr-x_shift)+a[2]+y_shift

def calcStrike(ballCoor, z_top, z_bottom):
  x_bt = ballCoor[0]
  y_bt = ballCoor[1]
  z_bt = ballCoor[2]

  x_left = -8.5 
  x_right = 8.5
  y_front = -8.5 
  y_back = 0
  y_veryback = 8.47
  bias = 8.47
  slope = -8.47/8.5

  if z_bt >= z_bottom and z_bt <= z_top:
    if x_bt >= x_left and x_bt <= x_right:
      if y_bt >= y_front and y_bt <= y_back:
          return True
      else: 
        if y_bt <= y_veryback:
          y_new = slope*x_bt + bias
          if y_bt <= y_new:
            print(y_new, y_bt)
            return True
  
  return False

def isStrike(ballCoors):
  strikeResults = []
  for ballCoor in ballCoors:
    strikeResults.append(calcStrike(ballCoor, max_height, min_height))
  return strikeResults
